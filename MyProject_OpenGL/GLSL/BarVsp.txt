// Vertex Shader
#version 460 compatibility


uniform vec2        uTexOffset;


// テクスチャ座標
out     vec2        vTexCoord;


void main( void )
{
    // 頂点計算
    gl_Position = gl_ModelViewProjectionMatrix * gl_Vertex;

    // テクスチャ座標セット
    vTexCoord = gl_MultiTexCoord0.st + uTexOffset;

}
