// Vertex Shader
#version 460 compatibility


// テクスチャ座標
out     vec2        vTexCoord;
out     vec4        vColor;


void main( void )
{
    // 頂点計算
    gl_Position = gl_ModelViewProjectionMatrix * gl_Vertex;

    // テクスチャ座標セット
    vTexCoord = gl_MultiTexCoord0.st;

    // カラーデータのセット
    vColor = gl_Color;

}
