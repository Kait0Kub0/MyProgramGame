// Vertex Shader
#version 460 compatibility

//毎フレームCPUから直接設定される
// uniform vec3 offset;

// 外部データの参照（アトリビュート）※頂点と同じ数必要
// 通常は、頂点バッファが使用される
// in vec3 dat;

// テクスチャ座標(バーリング)
out     vec2        vTexCoord;

void main( void )
{
    // 頂点計算
    gl_Position = gl_ModelViewProjectionMatrix * gl_Vertex;

    // テクスチャ座標セット
    vTexCoord = gl_MultiTexCoord0.st;

}
